  const promises = [];

  // Tile sprites
  for (const [tileId, path] of Object.entries(TILE_SPRITES)) {
    promises.push(
      loadImage(path)
        .then(img => tiles.set(Number(tileId), img))
        .catch(() => { /* fallback to color */ })
    );
  }

  // Object sprites
  for (const [visual, path] of Object.entries(OBJECT_SPRITE_PATHS)) {
    promises.push(
      loadImage(path)
        .then(img => objects.set(visual, img))
        .catch(() => { /* fallback to color */ })
    );
  }

  // Character sprites
  for (const [key, path] of Object.entries(CHARACTER_SPRITE_PATHS)) {
    promises.push(
      loadImage(path)
        .then(img => characters.set(key, img))
        .catch(() => { /* fallback to color */ })
    );

  // Prop sprites
  for (const [key, path] of Object.entries(PROP_SPRITES)) {
    promises.push(
      loadImage(path)
